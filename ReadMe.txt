# Mlflow installation (Works better with python 3.10)
pip install mlflow

# Launch the MLflow Tracking Server
mlflow server --host 127.0.0.1 --port 8080

# Start the MLflow UI
mlflow ui --host 127.0.0.1 --port 8090
